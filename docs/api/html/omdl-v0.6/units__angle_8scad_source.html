77 //! Return the name of an angle unit identifier.
78 /***************************************************************************//**
79  \param u <string> An angle unit identifier.
80 
81  \returns <string> The units name for the given angle unit identifier.
82  Returns \b undef for identifiers that are not defined.
83 *******************************************************************************/
84 function unit_angle_name
85 (
86  u = base_unit_angle
87 ) = u == "r" ? "radian"
88  : u == "d" ? "degree"
89  : u == "dms" ? "degree, minute, second"
90  : undef;
91 
92 //! Convert an angle from degrees to other units.
93 /***************************************************************************//**
94  \param a <decimal|decimal-list-3> An angle to convert.
95  \param to <string> The units to which the angle should be converted.
96 
97  \returns <decimal|decimal-list-3> The conversion result.
98  Returns \b undef for identifiers that are not defined.
99 
100  \private
101 *******************************************************************************/
102 function unit_angle_d_to
103 (
104  a,
105  to
106 ) = to == "r" ? (a * tau / 360)
107  : to == "d" ? (a)
108  : to == "dms" ? ([
109  floor(a),
110  floor((a - floor(a)) * 60),
111  (a - floor(a) - floor((a - floor(a)) * 60) / 60) * 3600
112  ])
113  : undef;
114 
115 //! Convert an angle from some units to degrees.
116 /***************************************************************************//**
117  \param a <decimal|decimal-list-3> An angle to convert.
118  \param from <string> The units of the angle to be converted.
119 
120  \returns <decimal|decimal-list-3> The conversion result.
121  Returns \b undef for identifiers that are not defined.
122  \private
123 *******************************************************************************/
124 function unit_angle_to_d
125 (
126  a,
127  from
128 ) = from == "r" ? (a * 360 / tau)
129  : from == "d" ? (a)
130  : from == "dms" ? (a[0] + a[1]/60 + a[2]/3600)
131  : undef;
132 
133 //! Convert an angle from some units to another.
134 /***************************************************************************//**
135  \param a <decimal|decimal-list-3> An angle to convert.
136  \param from <string> The units of the angle to be converted.
137  \param to <string> A units to which the angle should be converted.
138 
139  \returns <decimal|decimal-list-3> The conversion result.
140  Returns \b undef for identifiers that are not defined.
141 *******************************************************************************/
142 function convert_angle
143 (
144  a,
145  from = base_unit_angle,
146  to = base_unit_angle
147 ) = unit_angle_d_to( unit_angle_to_d( a, from ), to );
